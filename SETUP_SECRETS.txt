# Простейший способ настроить secrets
# Скопируй и выполни эти команды в PowerShell ОДНУ ЗА ДРУГОЙ:

# 1. Авторизуйся (один раз):
# gh auth login

# 2. После авторизации выполни эти команды:

gh secret set TEST_SERVER_HOST --body "88.210.35.183"
gh secret set TEST_SERVER_USER --body "root"

# Для SSH ключа:
ssh root@88.210.35.183 "cat ~/.ssh/github_actions_deploy" | gh secret set TEST_SERVER_SSH_KEY

# Проверка:
gh secret list
